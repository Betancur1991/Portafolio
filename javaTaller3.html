6-Se está creando una aplicación que va a estar conectada con un prototipo que permita almacenar contactos telefónicos en el dispositivo. Para ello cada contacto debe contener nombre completo, teléfono y organización. import java.util.List;
import java.util.Scanner;
public class Main {
public static void main(String[] args) {
Agenda miAgenda = new Agenda();

import java.util.ArrayList;
import java.util.List;
public class Agenda {
private ArrayList contactos;
private final int CAPACIDAD_MAXIMA = 3;
public Agenda() {
contactos = new ArrayList<>();
}
public boolean agregarContacto(Contactos contacto) {
for (Contactos c : contactos) {
if (c.getPhoneNumber().equals(contacto.getPhoneNumber())) {
System.out.println("Error: El número de teléfono ya está en uso por otro contacto.");
return false;
}
}
if (contactos.size() < CAPACIDAD_MAXIMA) {
contactos.add(contacto);
return true;
} else {
System.out.println("Error: La agenda está llena.");
return false;
}
}
public List buscarContacto(String nombre) {
List contactosEncontrados = new ArrayList<>();
for (Contactos contacto : contactos) {
if (contacto.getName().equals(nombre)) {
contactosEncontrados.add(contacto);
}
}
return contactosEncontrados;
}
public boolean eliminarContacto(int id) {
for (Contactos contacto : contactos) {
if (contacto.getId() == id) {
contactos.remove(contacto);
return true;
}
}
return false;
}
public void imprimirContactos() {
if (contactos.size() > 0) {
System.out.println("Lista de contactos:");
for (Contactos contacto : contactos) {
System.out.println("Id: "+contacto.getId()+" Nombre: " + contacto.getName() + ", Teléfono: " + contacto.getPhoneNumber() + ", Organización: " + contacto.getOrganizacion());
}
} else {
System.out.println("La agenda está vacía.");
}
}
}


public class Contactos {
private String name;
private String phoneNumber;
private String organizacion;
private int id;
private static int nextId = 1;
public Contactos(String name, String phoneNumber, String organizacion) {
this.name = name;
this.phoneNumber = phoneNumber;
this.organizacion = organizacion;
this.id = nextId;
nextId++;
}
public String getName() {
return name;
}
public String getPhoneNumber() {
return phoneNumber;
}
public String getOrganizacion() {
return organizacion;
}
public int getId() {
return id;
}
}

import java.util.ArrayList;
public class Parquedero {
private ArrayList vehiculos;
private final int CAPACIDAD_MAXIMA = 5;
public Parquedero() {
vehiculos = new ArrayList<>();
}
public boolean ingresarVehiculo(Vehiculo vehiculo) {
for (Vehiculo v : vehiculos) {
if (v.getPlaca().equals(vehiculo.getPlaca())) {
System.out.println("Error: El vehiculo que intenta ingresar ya se encuentra ingresado.");
return false;
}
}
if (vehiculos.size() < CAPACIDAD_MAXIMA) {
vehiculos.add(vehiculo);
return true;
} else {
System.out.println("Error: Lo sentimos el parqueadero se encuentra lleno.");
return false;
}
}
public boolean retirarVehiculo(String placa) {
for (Vehiculo v : vehiculos) {
if (v.getPlaca().equals(placa)) {
System.out.println("El vehiculo se ha retirado del parquedero.");
vehiculos.remove(v);
return true;
}else {
System.out.println("Error: Lo sentimos el vehículo no se encuentra en el parqueadero.");
return false;
}
}return false;
}
public boolean consultarVehiculo(String placa) {
for (Vehiculo v : vehiculos) {
if (v.getPlaca().equals(placa)) {
return true;
}
}
for (Vehiculo v : vehiculos) {
if (!v.getPlaca().equals(placa)) {
return false;
}
}
return false;
}
}

import java.util.ArrayList;
import java.util.Random;
public class Escuela {
private ArrayList< Usuario > usuarios;
private final int CAPACIDAD_MAXIMA = 8;
public Escuela(){
usuarios=new ArrayList<>();
}
public boolean agregarEstudiante(Usuario usuario){
for (Usuario u:usuarios){
if (u.getCedula().equals(usuario.getCedula())){
System.out.println("Error: ya existe un usuario con esta identificacion.");
return false;
}
}
if (usuarios.size()< CAPACIDAD_MAXIMA){
usuarios.add(usuario);
return true;
}else {
System.out.println("Error: Ya no tenemos mas cupos disponibles por favor espera a que abramos nuevas inscripciones.");
return false;
}
}
public boolean conocerResultados(String cedula){
Random random= new Random();
int nota=random.nextInt(10)+1;
for (Usuario u:usuarios){
if(u.getCedula().equals(cedula)){
if (nota< 5){
System.out.println("Has perdido el curso, vuelve a intentarlo");
return true;
}else if (nota>=5){
System.out.println("Felicidades has aprobado el curso");
return true;
}
}
}return false;
}
public void mostrarUsuariosRegistrados(){

}
}
