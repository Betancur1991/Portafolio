this.estatura = estatura;

return resultado;

} else if (calculoIMC>25 && calculoIMC<=29.9) {
resultado=fullName+" Estas en sobrepeso, debes mejorar tu alimentación";
return resultado;
} else if (calculoIMC>30 && calculoIMC<=34.9) {
resultado=fullName+" Estas en obesidad grado 1, debes mejorar tu alimentación ya que puedes tener problemas de salud";
return resultado;
}else if (calculoIMC>35 && calculoIMC<=39.9) {
resultado=fullName+ "Estas en obesidad grado 2, debes mejorar tu alimentación ya que puedes tener problemas de salud";
return resultado;
}else{ resultado=fullName+" Estas en obesidad grado 3, debes buscar ayuda medica ya que tu grado de obesidad puede tener concecuencias fatales";
return resultado;
}

public class Figura {

this.altura = altura;
this.baseB = baseB;
}

return baseA;
}
public double getAltura() {
return altura;
}
public double getBaseB() {
return baseB;
}

this.baseA = baseA;
this.altura = altura;
}
public String calcularAreaTriangulo(){
double areaTriangulo=(baseA*altura)/2;

double areaRectangulo=(baseA*altura);
return "El área del rectangulo ingresado es de: "+areaRectangulo+" metros";
}
public String calcularAreaTrapecio(){
double areaTrapecio=(baseA+baseB)*altura/2;

import java.util.ArrayList;
public class Banco {
private ArrayList cuentas;
public Banco() {
cuentas = new ArrayList();
}
public void agregarCuenta(Cuenta cuenta) {
cuentas.add(cuenta);
}
public Cuenta buscarCuenta(int numeroCuenta) {
for (Cuenta cuenta : cuentas) {
if (cuenta.getNumeroCuenta() == numeroCuenta) {
return cuenta;
}
}
return null;
}
}


public class Cuenta {
